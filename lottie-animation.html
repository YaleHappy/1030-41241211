Lottie動畫

Lottie.js官網

程式碼範例

<!-- Lottie.js -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.7.13/lottie.min.js"></script>

<!-- Lottie 動畫初始化 -->
<script>
    lottie.loadAnimation({
        container: document.getElementById('lottie-animation'), // 動畫容器
        renderer: 'svg', // 渲染方式
        loop: true, // 是否循環播放
        autoplay: true, // 自動播放
        path: 'YOUR_JSON_FILE_URL.json' // JSON 動畫路徑
    });
</script>

<!-- 動畫顯示區域 -->
<div id="lottie-animation" style="width: 300px; height: 300px;"></div>
