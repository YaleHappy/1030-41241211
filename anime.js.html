Anime.js 動畫效果

Anime.js 官網

程式碼範例

<!-- 引入 Anime.js CDN -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/animejs/3.2.1/anime.min.js"></script>

<!-- 動畫區塊 -->
<div class="animation-box" id="animation-box"></div>

<!-- 初始化 Anime.js 動畫 -->
<script>
    document.addEventListener("DOMContentLoaded", function () {
        anime({
            targets: '#animation-box',
            translateX: 250,    // 向右移動 250px
            rotate: '1turn',    // 旋轉一圈
            backgroundColor: '#ff6f61',  // 改變背景色
            duration: 2000,     // 持續時間 2000 毫秒
            easing: 'easeInOutQuad'  // 緩動效果
        });
    });
</script>
