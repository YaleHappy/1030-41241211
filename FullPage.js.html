FullPage.js 滾動效果

FullPage.js 官網

程式碼範例

<!-- FullPage.js CDN -->
<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/fullPage.js/3.1.2/fullpage.min.css">
<script src="https://cdnjs.cloudflare.com/ajax/libs/fullPage.js/3.1.2/fullpage.min.js"></script>

<!-- FullPage.js 初始化 -->
<script>
    document.addEventListener("DOMContentLoaded", function () {
        new fullpage('#fullpage', {
            autoScrolling: true,
            navigation: true,
        });
    });
</script>

<!-- FullPage.js 區塊 -->
<div id="fullpage">
    <div class="section section-1">頁面 1</div>
    <div class="section section-2">頁面 2</div>
    <div class="section section-3">頁面 3</div>
    <div class="section section-4">頁面 4</div>
</div>
