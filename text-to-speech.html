文字轉語音

Web Speech API官網

程式碼範例

            <!-- HTML 內容 -->
            <div class="mt-4">
                <label for="text-input" class="form-label">請輸入文字:</label>
                <input type="text" id="text-input" class="form-control" placeholder="輸入文字讓我讀給你聽">
                <button class="btn btn-primary mt-3" onclick="speakText()">播放語音</button>
            </div>
            
            <!-- JavaScript -->
            <script>
                function speakText() {
                    const text = document.getElementById('text-input').value;
                    if (text) {
                        const utterance = new SpeechSynthesisUtterance(text);
                        utterance.lang = 'zh-TW';  // 設置語言（例如：中文）
                        window.speechSynthesis.speak(utterance);
                    } else {
                        alert("請輸入文字！");
                    }
                }
            </script>
